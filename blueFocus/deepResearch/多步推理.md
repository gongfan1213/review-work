要实现“多步推理”功能的详细代码文件，核心思想是：**将复杂问题自动拆解为多个子问题，逐步检索、反思、聚合，最终输出高质量答案**。这类多步推理 agent 的典型结构包括：

1. 问题分析与分解（自动生成子问题）
2. 针对每个子问题进行检索/推理
3. 动态评估每步结果，必要时反思或细化
4. 聚合所有子问题的答案，形成最终结论

下面给出一个简化版的“多步推理 Agent”Python 代码示例，适合教学和二次开发。你可以将其保存为 `multi_step_reasoning_agent.py`：

```python
import openai

class MultiStepReasoningAgent:
    def __init__(self, openai_api_key, model="gpt-3.5-turbo"):
        openai.api_key = openai_api_key
        self.model = model

    def ask_llm(self, prompt, temperature=0.2):
        response = openai.ChatCompletion.create(
            model=self.model,
            messages=[{"role": "user", "content": prompt}],
            temperature=temperature,
        )
        return response.choices[0].message["content"].strip()

    def decompose_question(self, question):
        prompt = (
            f"请将以下复杂问题拆解为若干个有逻辑顺序的子问题，每个子问题独立成行：\n问题：{question}\n子问题："
        )
        subquestions = self.ask_llm(prompt)
        return [q.strip() for q in subquestions.split("\n") if q.strip()]

    def solve_subquestion(self, subquestion, context=None):
        prompt = f"请详细回答子问题：{subquestion}"
        if context:
            prompt += f"\n已知信息：{context}"
        return self.ask_llm(prompt)

    def aggregate_answers(self, question, subqa_pairs):
        context = "\n".join([f"子问题：{q}\n答案：{a}" for q, a in subqa_pairs])
        prompt = (
            f"请根据以下子问题及其答案，综合推理并给出对原始问题“{question}”的最终详细解答：\n{context}\n最终答案："
        )
        return self.ask_llm(prompt)

    def multi_step_reasoning(self, question):
        # 步骤1：问题分解
        subquestions = self.decompose_question(question)
        subqa_pairs = []
        # 步骤2：逐步推理
        for subq in subquestions:
            answer = self.solve_subquestion(subq)
            subqa_pairs.append((subq, answer))
        # 步骤3：答案聚合
        final_answer = self.aggregate_answers(question, subqa_pairs)
        return {
            "question": question,
            "subquestions": subqa_pairs,
            "final_answer": final_answer
        }

if __name__ == "__main__":
    # 用你的 OpenAI API Key 替换下面的字符串
    agent = MultiStepReasoningAgent(openai_api_key="YOUR_OPENAI_API_KEY")
    question = "为什么人类要探索太空？"
    result = agent.multi_step_reasoning(question)
    print("原始问题：", result["question"])
    print("\n【多步推理过程】")
    for idx, (subq, ans) in enumerate(result["subquestions"], 1):
        print(f"第{idx}步：{subq}\n答：{ans}\n")
    print("【最终综合答案】\n", result["final_answer"])
```
