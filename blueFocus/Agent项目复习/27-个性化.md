    """动态识别上一步的内容并格式化为引用内容"""
    
    # 处理用户选择的画像
    profiles = nova3_selections.get("nova3_profile", [])
    if profiles:
        profile_texts = []
        for profile in profiles:
            if profile.get("selected", True):
                text = f"@{profile['id']}: {profile['content']}"
                if profile.get("user_comment"):
                    text += f" [用户评论: {profile['user_comment']}]"
                profile_texts.append(text)
        if profile_texts:
            referenced_parts.append("受众画像:\n" + "\n".join(profile_texts))
