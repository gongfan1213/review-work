            'normal': {
                'max_connections': 100,  # 中等连接数
                'timeout': 10            # 中等超时
            },
            'background': {
                'max_connections': 50,   # 较少连接
                'timeout': 15            # 较长超时
            }
        }
    
    async def get_redis_client(self, pool_type: str = 'high_priority'):
        """获取Redis客户端，支持不同优先级的连接池"""
        # 先检查连接池是否已存在
        if pool_type in self._redis_pools:
            return self._redis_pools[pool_type]
        
        # 创建新的连接池
        config = self._pool_configs[pool_type]
        pool = aioredis.ConnectionPool.from_url(
            redis_url,
            max_connections=config['max_connections'],
            retry_on_timeout=True,
            socket_timeout=config['timeout']
        )
        
        self._redis_pools[pool_type] = aioredis.Redis(connection_pool=pool)
        return self._redis_pools[pool_type]
```

async def _detect_disconnect(self) -> bool:
    """检测用户是否断网（增强版：优化日志输出）"""
    try:
        # 获取用户最后心跳时间
        last_heartbeat = await persistence_manager.stream_storage.get_last_heartbeat(
            self._current_user_id, self._current_session_id
        )
        
        # 🔧 修复时区问题：使用UTC时间确保与数据库时间一致
        current_time = datetime.now(timezone.utc).replace(tzinfo=None)
        disconnect_threshold = current_time - timedelta(seconds=30)
        
        # 如果超过30秒没有心跳，认为可能断网了
        is_disconnected = last_heartbeat < disconnect_threshold
        
        if is_disconnected:
            seconds_since_heartbeat = (current_time - last_heartbeat).total_seconds()
            
            # 🔧 优化：使用去重机制，避免频繁输出断网日志
            session_key = f"{self._current_user_id}:{self._current_session_id}"
            current_timestamp = time.time()
            last_log_time = self._last_disconnect_log_time.get(session_key, 0)
            
            # 只有距离上次输出断网日志超过指定间隔才输出
            if current_timestamp - last_log_time > self._disconnect_log_interval:
                self.logger.info(f"🚨 [断网检测] 检测到用户断网! 上次心跳距今{seconds_since_heartbeat:.1f}秒")
                self._last_disconnect_log_time[session_key] = current_timestamp
            else:
                # 在间隔内的重复检测使用DEBUG级别
                self.logger.debug(f"🔍 [断网检测] 用户仍处于断网状态，上次心跳距今{seconds_since_heartbeat:.1f}秒")
        
        return is_disconnected
        
    except Exception as e:
        self.logger.error(f"断网检测失败: {e}")
        return False
```

**设计模式**：
- **心跳检测模式**：定期检查用户连接状态
- **状态机模式**：断网/在线状态转换
- **日志去重模式**：避免重复日志输出
- **容错模式**：检测失败时优雅降级
