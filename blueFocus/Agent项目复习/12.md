def parse_sse_stream(self, response, request_id: str, user_id: str, live_file_path: str) -> tuple:
    """解析Server-Sent Events (SSE)流，支持实时输出"""
    content_chunks = []
    start_time = datetime.now()
    
    with open(live_file_path, 'w', encoding='utf-8') as live_file:
        line_count = 0
        for line in response.iter_lines(decode_unicode=True):
            if line:
                line_count += 1
                timestamp = datetime.now().strftime('%H:%M:%S.%f')[:-3]
                
                # SSE格式通常是 "data: {json_data}"
                if line.startswith('data: '):
                    data_line = line[6:]  # 去掉 "data: " 前缀
                    content_chunks.append(data_line)
                    
                    # 实时控制台输出
                    if line_count <= 5 or line_count % 10 == 0:
                        print(f"📡 [{request_id}] [{timestamp}] {data_line[:100]}...")
                    
                    # 实时写入文件
                    live_file.write(f"[{timestamp}] {data_line}\n")
                    live_file.flush()  # 确保实时写入
    
    return '\n'.join(content_chunks), start_time, end_time
