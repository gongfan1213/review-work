```python
class LayeredQueueManager:
    """
    三层队列管理器
    
    存储层级：
    L1: 内存缓存 (最快访问，容量有限)
    L2: Redis缓存 (分布式共享，中等速度)  
    L3: 本地文件 (持久化存储，容量大)
    
    工作流程：
    - 读取：内存 -> Redis -> 文件
    - 写入：同时写入所有层级
    - 清理：定期清理过期数据
    """
    
    async def save_queue(self, user_id: str, session_id: str, queue_data: Any):
        """保存队列 - 写入所有层级，并发写入提升性能"""
        queue_key = self._generate_queue_key(user_id, session_id)
        
        # L1: 立即写入内存
        self._save_to_memory(queue_key, queue_data)
        
        # L2 & L3: 并发写入Redis和文件
        tasks = []
        tasks.append(self._save_to_redis(queue_key, queue_data))
        
        # 文件写入在线程池中执行，避免阻塞
        loop = asyncio.get_event_loop()
        tasks.append(loop.run_in_executor(None, self._save_to_file, queue_key, queue_data))
        
        # 等待所有写入完成
        await asyncio.gather(*tasks, return_exceptions=True)
```
