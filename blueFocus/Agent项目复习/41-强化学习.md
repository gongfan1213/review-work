### 1. 高层决策RL - Meta-Orchestrator

**目标**：学习最优的任务分解和Agent选择策略

```python
"""
RL-Enhanced Meta-Orchestrator
使用强化学习优化任务分解和Agent选择
"""
import torch
import torch.nn as nn
import torch.optim as optim
from torch.distributions import Categorical
import numpy as np
from typing import Dict, List, Any, Tuple
import json

class MetaOrchestratorRL(BaseLoomiAgent):
    """
    基于强化学习的元编排器
    学习如何最优地分解任务和选择Agent
    """
    
    def __init__(self):
        super().__init__("meta_orchestrator_rl")
        
        # RL模型组件
        self.policy_network = TaskDecompositionPolicy()
        self.value_network = TaskValueEstimator()
        self.optimizer = optim.Adam(
            list(self.policy_network.parameters()) + 
            list(self.value_network.parameters()),
            lr=0.001
        )
        
        # 经验回放缓冲区
        self.experience_buffer = ExperienceReplayBuffer(max_size=10000)
        
        # 可用的Agent类型
        self.available_agents = [
            'hitpoint', 'persona', 'knowledge', 'resonant',
            'websearch', 'brand_analysis', 'content_analysis',
            'xhs_post', 'tiktok_script', 'wechat_article', 'revision'
        ]
        
        # 任务类型编码
        self.task_types = {
            'content_creation': 0,
            'market_analysis': 1,
            'user_research': 2,
            'content_optimization': 3,
            'strategy_planning': 4
        }
        
        self.logger.info("🧠 Meta-Orchestrator RL初始化完成")
    
    async def process_request(self, request_data: Dict[str, Any], context: Optional[Dict[str, Any]] = None):
        """RL增强的任务处理流程"""
        user_id = request_data.get("user_id", "unknown")
        session_id = request_data.get("session_id", "unknown")
        query = request_data.get("query", "")
        
        # 设置当前会话
        self.set_current_session(user_id, session_id)
        
        # 1. 状态编码
        state = await self._encode_state(user_id, session_id, query)
        
        # 2. RL策略决策
        action_probs, value_estimate = await self._rl_decision(state)
        
        # 3. 执行决策
        selected_agents, execution_plan = await self._execute_rl_plan(
            action_probs, user_id, session_id, query
        )
        
        # 4. 记录经验用于学习
        await self._record_experience(state, action_probs, value_estimate)
        
        # 5. 流式返回结果
        async for event in self._stream_execution_results(selected_agents, execution_plan):
            yield event
    
    async def _encode_state(self, user_id: str, session_id: str, query: str) -> torch.Tensor:
        """将当前状态编码为RL模型输入"""
        # 获取用户历史偏好
        user_preferences = await self._get_user_preferences(user_id)
        
        # 获取会话上下文
        session_context = await self._get_session_context(user_id, session_id)
        
        # 查询特征提取
        query_features = await self._extract_query_features(query)
        
        # 组合状态向量
        state_vector = np.concatenate([
            user_preferences,
            session_context,
            query_features
        ])
        
        return torch.FloatTensor(state_vector)
    
    async def _rl_decision(self, state: torch.Tensor) -> Tuple[torch.Tensor, torch.Tensor]:
        """使用RL模型进行决策"""
        with torch.no_grad():
            # 策略网络输出动作概率
            action_logits = self.policy_network(state)
            action_probs = torch.softmax(action_logits, dim=-1)
            
            # 价值网络输出状态价值
            value_estimate = self.value_network(state)
            
        return action_probs, value_estimate
    
    async def _execute_rl_plan(self, action_probs: torch.Tensor, user_id: str, session_id: str, query: str):
        """执行RL决策的Agent选择计划"""
        # 根据概率选择Agent
        selected_agents = []
        execution_plan = []
        
        # 选择概率最高的前几个Agent
        top_k = 3
        _, top_indices = torch.topk(action_probs, top_k)
        
        for idx in top_indices:
            agent_name = self.available_agents[idx.item()]
            selected_agents.append(agent_name)
            
            # 构建执行计划
            execution_plan.append({
                'agent': agent_name,
                'priority': float(action_probs[idx]),
                'estimated_time': self._estimate_execution_time(agent_name),
                'expected_output': self._get_expected_output_type(agent_name)
            })
        
        return selected_agents, execution_plan
    
    async def _record_experience(self, state: torch.Tensor, action_probs: torch.Tensor, value_estimate: torch.Tensor):
        """记录经验到回放缓冲区"""
        experience = {
            'state': state,
            'action_probs': action_probs,
            'value_estimate': value_estimate,
            'timestamp': datetime.now().isoformat()
        }
        
        self.experience_buffer.add(experience)
    
    async def train_rl_model(self, batch_size: int = 32):
        """训练RL模型"""
        if len(self.experience_buffer) < batch_size:
            return
        
        # 采样经验批次
        batch = self.experience_buffer.sample(batch_size)
        
        # 计算损失
        policy_loss = self._compute_policy_loss(batch)
        value_loss = self._compute_value_loss(batch)
        
        total_loss = policy_loss + value_loss
        
        # 反向传播
        self.optimizer.zero_grad()
        total_loss.backward()
        self.optimizer.step()
        
        self.logger.info(f"🧠 RL模型训练完成: policy_loss={policy_loss:.4f}, value_loss={value_loss:.4f}")


class TaskDecompositionPolicy(nn.Module):
    """任务分解策略网络"""
    
    def __init__(self, input_dim: int = 128, hidden_dim: int = 256, output_dim: int = 11):

        self.network = nn.Sequential(

            nn.Dropout(0.2),

    def forward(self, state: torch.Tensor) -> torch.Tensor:
        return self.network(state)


class TaskValueEstimator(nn.Module):
    """任务价值估计网络"""
    
    def __init__(self, input_dim: int = 128, hidden_dim: int = 256):
        super().__init__()
        
        self.network = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Dropout(0.2),

    def forward(self, state: torch.Tensor) -> torch.Tensor:
        return self.network(state)


class ExperienceReplayBuffer:
    """经验回放缓冲区"""
    
    def __init__(self, max_size: int = 10000):
        self.buffer = []
        self.max_size = max_size
    
    def add(self, experience: Dict[str, Any]):
        """添加经验"""
        self.buffer.append(experience)
        if len(self.buffer) > self.max_size:
            self.buffer.pop(0)
    
    def sample(self, batch_size: int) -> List[Dict[str, Any]]:
        """采样经验批次"""
        import random
        return random.sample(self.buffer, min(batch_size, len(self.buffer)))
    
    def __len__(self):
        return len(self.buffer)
```

```python
                yield event
        else:
```
