class LayeredQueueManager:
    """
    三层队列管理器 - 隐式优先级调度
    
    存储层级（按访问优先级）：
    L1: 内存缓存 (最高优先级，最快访问，容量有限)
    L2: Redis缓存 (中等优先级，分布式共享，中等速度)  
    L3: 本地文件 (最低优先级，持久化存储，容量大)

        self.memory_cache_ttl_hours = 1  # 内存缓存TTL
        self.redis_ttl_hours = 24       # Redis缓存TTL
        self.file_retention_hours = 168  # 文件保留时间（7天）

#### **读取优先级调度**
```python
async def get_queue(self, user_id: str, session_id: str) -> Optional[Any]:
    """获取队列 - 按优先级层级查找"""
    queue_key = self._generate_queue_key(user_id, session_id)
    
    # 🔥 L1: 最高优先级 - 内存缓存
    queue_data = self._get_from_memory(queue_key)
    if queue_data is not None:
        return queue_data  # 立即返回，最快访问
    
    # 🔥 L2: 中等优先级 - Redis缓存
    queue_data = await self._get_from_redis(queue_key)
    if queue_data is not None:
        # 回写到内存，提升下次访问优先级
        self._save_to_memory(queue_key, queue_data)
        return queue_data
    
    # 🔥 L3: 最低优先级 - 文件存储
    queue_data = self._get_from_file(queue_key)
    if queue_data is not None:
        # 回写到上层缓存，提升访问优先级
        self._save_to_memory(queue_key, queue_data)
        await self._save_to_redis(queue_key, queue_data)
        return queue_data
    
    return None
```

### **3. 任务队列的隐式优先级调度**

```python
class TaskQueue:
    """任务队列 - 基于时间顺序的优先级"""
    
    def get_current_task(self) -> Optional[TaskItem]:
        """获取当前任务 - 按创建时间优先级"""
        self.last_accessed = datetime.now().isoformat()
        if self.current_task_index < len(self.tasks):
            return self.tasks[self.current_task_index]  # FIFO优先级
        return None
    
    def mark_current_completed(self, result: str = ""):
        """标记当前任务完成，自动调度下一个"""
        if self.current_task_index < len(self.tasks):
            task = self.tasks[self.current_task_index]
            task.status = "completed"
            task.result = result
            task.completed_at = datetime.now().isoformat()
            self.current_task_index += 1  # 移动到下一个优先级任务
```

### **4. 缓存优先级管理**

#### **LRU优先级淘汰策略**
```python
def _cleanup_memory_cache(self):
    """清理内存缓存 - 基于LRU优先级"""
    current_time = datetime.now().timestamp()
    expired_keys = []
    
    # 第一优先级：删除过期项
    for key, item in self.memory_cache.items():
        if item.get("expires_at", 0) < current_time:
            expired_keys.append(key)
    
    # 删除过期项
    for key in expired_keys:
        del self.memory_cache[key]
    
    # 第二优先级：如果还是太多，删除最老访问的项
    if len(self.memory_cache) >= self.max_memory_cache_size:
        sorted_items = sorted(
            self.memory_cache.items(),
            key=lambda x: x[1].get("last_accessed", 0)  # 按访问时间排序
        )
        
        # 删除最老的一半（LRU策略）
        for key, _ in sorted_items[:len(sorted_items)//2]:
            del self.memory_cache[key]
```

async def process_request(self, request_data: Dict[str, Any], context: Optional[Dict[str, Any]] = None):
    """处理用户请求 - 系统级优先级调度"""

    if not self._validate_request_data(request_data):
        yield await self._emit_event(EventType.ERROR, data="请求数据无效")
        return

    if query and self.resolver.should_resolve(query):
        resolved_query = await self.resolver.resolve_references(query, user_id, session_id)
        if resolved_query != query:
            request_data["query"] = resolved_query
    
    # 3. 最高优先级：检查现有队列
    current_queue = await self._get_queue_safe(user_id, session_id)
    if current_queue and current_queue.has_pending_tasks():

    else:
        # 生成新计划（中等优先级）

### **7. 优先级调度的设计亮点**

#### **🔥 多层优先级架构**
1. **存储优先级**：内存 > Redis > 文件
2. **连接池优先级**：high_priority > normal > background  
3. **任务优先级**：FIFO + 时间戳
4. **Agent优先级**：核心Agent > 普通Agent

#### **🚀 智能优先级调度**
1. **自适应调度**：根据访问频率自动提升优先级
2. **降级机制**：高优先级失败时自动降级
3. **并发控制**：通过信号量控制资源竞争
4. **缓存优化**：LRU策略保证热点数据优先级

#### **⚡ 性能优化优先级**
1. **预加载**：高优先级数据预先加载到内存
2. **预热**：连接池预热提升响应速度
3. **并发**：Comfort消息和主逻辑并发执行
4. **流式输出**：实时响应，避免阻塞

## 🎯 **总结**

这个系统的优先级调度虽然没有显式的优先级队列，但通过以下机制实现了高效的优先级调度：

1. **三层存储架构**：内存 > Redis > 文件，按访问速度分层

6. **自适应调度**：根据访问模式动态调整优先级

这种设计既保证了系统的响应性能，又确保了资源的合理分配，是一个非常优雅的优先级调度实现。
