-- schema 可选：使用 public
create table if not exists public.graph_checkpoints (
  id bigserial primary key,
  thread_id text not null,
  checkpoint_ns text not null default '',
  checkpoint_id text not null,
  parent_checkpoint_id text null,
  user_id text not null,
  session_id text not null,
  checkpoint_type text not null,
  -- 代码里以 base64/text 形式存储；这里用 text，便于跨平台序列化
  checkpoint_data text not null,
  metadata_data jsonb not null default '{}'::jsonb,
  channel_versions jsonb not null default '{}'::jsonb,
  redis_key text null,
  redis_ttl_expires_at timestamptz null,
  backup_status text not null default 'active',
  is_active boolean not null default true,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

-- 唯一约束：与代码 upsert onConflict 对齐
create unique index if not exists ux_graph_checkpoints_tid_ns_id
  on public.graph_checkpoints (thread_id, checkpoint_ns, checkpoint_id);

-- 检索索引（可选）
create index if not exists ix_graph_checkpoints_thread_ns_created
  on public.graph_checkpoints (thread_id, checkpoint_ns, created_at desc);

-- 写入记录表
create table if not exists public.graph_checkpoint_writes (
  id bigserial primary key,
  checkpoint_ref_id bigint not null references public.graph_checkpoints(id) on delete cascade,
  thread_id text not null,
  checkpoint_ns text not null default '',
  checkpoint_id text not null,
  task_id text not null,
  write_idx int null,
  channel text not null,
  write_type text not null,
  -- 同样用 text 存储 base64
  write_data text not null,
  created_at timestamptz not null default now()
);

create index if not exists ix_graph_checkpoint_writes_refid_idx
  on public.graph_checkpoint_writes (checkpoint_ref_id, write_idx);
