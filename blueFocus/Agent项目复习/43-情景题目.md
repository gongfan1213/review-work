    """异步流生成器 - 零轮询，事件驱动"""
    while True:
        try:
            # 使用wait_for避免无限等待，1秒超时
            response_data = await asyncio.wait_for(
                session_info.response_queue.get(),
                timeout=1.0
            )

            if response_data['type'] == 'stop':
                yield f"{json.dumps(response_data, ensure_ascii=False)}\n\n"
                break
            elif response_data['type'] == 'chunk':
                # 流式数据块处理

                yield f"{json.dumps(formatted_chunk, ensure_ascii=False)}\n\n"
