封装指定客户经营接口人

角色职责： 进入【客户经营】页面管理客户各项服务的续费情况，及时调整续费跟进状态，并记录跟进情况。表格排序逻辑：排序：

1.主账号

2.服务总接口人

3.客户经营接口人

注意：服务总接口人和客户经营接口人可以指定同一个子账号

更改内容后提交时，二次确认，确认变更客户经营接口人？已选子账号后，激活"清空"按钮，点击后清空已选项，内容为空时提交弹窗，出现如下二次确认框
# 指定客户经营接口人一次弹窗代码

```js

const { success: successMessage } = useMessage();

export function customManagerPerson(onComplete?: () => void) {
  const { open, confirm } = useModal();
  const contentRef = ref();
  contentRef.value = h(DispatchServiceForm, { ref: 'dispatchServiceForm' });
  const successHandle = () => {
    successMessage($i('partner.production.module.modify_success'));
    onComplete?.();
    dispatchServiceWin.close();
  };

  const updateConfirm = (ids: string[]) => {
    confirm({
      title: $i('partner.customer_operation.customer_interface_change_confirm'),
      content: $i(
        'partner.customer_operation.customer_interface_change_confirm_content'
      ),
      onOk: async () => {
        const { success } = await putBusinessContact({
          data: {
            ids,
          },
        });
        success && successHandle();
        return success;
      },
    });
  };

  const deleteConfirm = (ids: string[]) => {
    confirm({
      title: $i('partner.customer_operation.customer_interface_delete_confirm'),
      content: $i(
        'partner.customer_operation.customer_interface_delete_confirm_content'
      ),
      onOk: async () => {
        const { success } = await deleteBusinessContact({
          data: {
            ids,
          },
        });
        success && successHandle();
        return success;
      },
    });
  };

  const dispatchServiceWin = open({
    header: $i('partner.customer_operation.designate_customer_manager'),
    width: 480,
    content: contentRef.value,
    destroyOnHide: true,
    onOk: async () => {
      const formRoot = contentRef.value.context.$refs.dispatchServiceForm;
      const isValid = formRoot?.isValid();
      const { ids, oldIds } = formRoot?.getJsonValue();
      if (isValid) {
        if (!ids.length && oldIds.length) {
          deleteConfirm(ids);
          return false;
        } else if (ids.length && oldIds.length) {
          updateConfirm(ids);
          return false;
        } else if (ids.length && !oldIds.length) {
          const { success } = await postBusinessContact({
            data: {
              ids,
            },
          });
          success && successHandle();
          return success;
        } else {
          dispatchServiceWin.close();
        }
      }
    },
  });
}
```

# 指定客户经营接口人二次弹窗代码
```js
<template>
  <div>
    <IxAlert type="offline" class="modal-alert">
      <div>
        <div>
          {{ $i('partner.customer_operation.customer_manager_duty') }}
        </div>
        <div>
          {{ $i('partner.customer_operation.customer_manager_duty_0') }}
        </div>
      </div>
    </IxAlert>
    <IxForm labelAlign="start" :control="formGroup" :labelCol="5" colonless>
      <IxFormItem :label="$i('partner.customer_operation.customer_manager')">
        <IxRow class="subaccount-item-container" gutter="8">
          <IxCol span="21">
            <IxSelect
              multiple
              control="ids"
              :placeholder="$i('partner.production.module.select_subaccount')"
              :dataSource="dataSource"
            />
          </IxCol>
          <IxCol span="2">
            <IxButton
              mode="link"
              size="xs"
              :disabled="isDeleteDisabled"
              @click="deleteHandle"
            >
              {{ $i('partner.production.module.service_interface_delete') }}
            </IxButton>
          </IxCol>
        </IxRow>
      </IxFormItem>
    </IxForm>
  </div>
</template>

const dataSource = ref<SelectData[]>([]);
const oldIds = ref<string[]>([]);
const formGroup = useFormGroup<{
  ids: string[];
}>({
  ids: [],
});

const isDeleteDisabled = computed(() => {
  return !formGroup.getValue()?.ids?.length;
});

const loadSubAccount = async () => {
  const { success, data } = await getSubAccount();

  if (success) {
    dataSource.value = (data?.data || []).map(
    (item: SubAccountData) => ({
      label: item.username,
      key: item.id,
    })
  );
  }
};

const getSubAdminName = async () => {
  let { success, data } = await getBusinessContact();
  if (success) {
    oldIds.value = (data?.data ?? []).map((item) => item.id);
    formGroup.setValue({
      ids: oldIds.value,
    });
  }
};

const deleteHandle = () => {
  formGroup.setValue({
    ids: []
  })
};

onMounted(async () => {
  loadSubAccount();
  getSubAdminName();
});

const getJsonValue = () => {
  return {
    ids: formGroup.getValue().ids,
    oldIds: oldIds.value,
  }
};

const isValid = () => {
  if (formGroup.valid.value) {
    return true;
  } else {
    formGroup.markAsDirty();
    return false;
  }
};

defineExpose({
  getJsonValue,
  isValid,
});
</script>
```

<div class="modal-alert">
    <IxAlert type="offline" class="modal-alert">
      <div>
        <div>
          {{ $i('partner.customer_operation.customer_manager_duty') }}
        </div>
        <div class="ccode_multiline-ellipse box-summary">
          {{ $i('partner.customer_operation.customer_manager_duty_0') }}
        </div>

    </IxAlert>

        <IxRow class="subaccount-item-container" gutter="8">

            <IxSelect
              multiple
              control="id"
              :placeholder="$i('partner.production.module.select_subaccount')"
              :dataSource="serviceInterfacePersonDataSource"
              :onChange="onUsernameChange"
            >
              <template #selectedLabel="option">
                {{ isValidKey(option.key) ? option.label : '' }}
              </template>
            </IxSelect>
          </IxCol>
          <IxCol span="2">
            <IxButton
              mode="link"
              size="xs"
              :disabled="isDeleteDisabled"
              @click="deleteInterfacePerson"
            >
              {{ $i('partner.production.module.service_interface_delete') }}
            </IxButton>
          </IxCol>
        </IxRow>
      </IxFormItem>
    </IxForm>
  </div>

const modal = useModal();
const serviceInterfacePersonDataSource = ref<SelectData[]>([]);
const oldId = ref<number | undefined>(undefined); // 保留回显的账号
const formGroup = useFormGroup<{
  id?: number;
}>({
  id: [undefined],
});

const isDeleteDisabled = computed(() => {
  return !formGroup.getValue().id;
});

const isValidKey = (key: number) => {
  const index = serviceInterfacePersonDataSource.value.findIndex(
    (item) => item.key === key
  );
  return index !== -1;
};
const loadSubAccount = async () => {
  const { success, data } = await getSubAccount({
    mask: false,
  });

  if (!success || !data?.data) {
    return;
  }

  serviceInterfacePersonDataSource.value = (data?.data || []).map(
    (item: SubAccountData) => ({
      label: item.username,
      key: item.id,
    })
  );
};

const getSubAdminName = async () => {
  let { success, data } = await getBusinessContact();
  if (!success || !data || !isValidKey(parseInt(data.id))) {
    return;
  }

  oldId.value = parseInt(data.id);
  formGroup.setValue({
    id: parseInt(data.id),
  });
};
const openModal = (
  isEmpty: boolean,
  value: number | undefined,
  oldValue: number | undefined
) =>
  modal.confirm({
    title: isEmpty
      ? $i('partner.customer_operation.customer_interface_delete_confirm')
      : $i('partner.customer_operation.customer_interface_change_confirm'),
    content: isEmpty
      ? $i(
          'partner.customer_operation.customer_interface_delete_confirm_content'
        )
      : $i(
          'partner.customer_operation.customer_interface_change_confirm_content'
        ),
    onOk: () => {
      formGroup.setValue({
        id: value,
      });
    },
    onCancel: () => {
      formGroup.setValue({
        id: oldValue,
      });
    },
  });

const onUsernameChange = (value: number, oldValue: number) => {
  // 非变更性指定接口人，不需要二次确认弹窗
  if (!oldId.value) {
    return;
  }
  openModal(!value, value, oldValue);
};

const deleteInterfacePerson = () => {
  const id = formGroup.getValue().id;
  openModal(true, undefined, id);
};

onMounted(async () => {
  await loadSubAccount();
  getSubAdminName();
});

const getJsonValue = () => {
  const id = formGroup.getValue().id;
  const username = serviceInterfacePersonDataSource.value.find(
    (item) => item.key === id
  )?.label;
  return {
    username,
    id,
  };
};

const isValid = () => {
  if (formGroup.valid.value) {
    return true;
  } else {
    formGroup.markAsDirty();
    return false;
  }
};

defineExpose({
  getJsonValue,
  isValid,
});
</script>

function dispatchCustomManagerPerson(
  onComplete?: (args: { success: boolean; isCancel?: boolean }) => void
) {
  const contentRef = ref();
  contentRef.value = h(DispatchServiceForm, { ref: 'dispatchServiceForm' });
  const { open, error } = useModal();
  const dispatchServiceWin = open({
    header: $i('partner.customer_operation.designate_customer_manager'),
    width: 480,
    content: contentRef.value,
    destroyOnHide: true,
    onOk: async () => {
      const formRoot = contentRef.value.context.$refs.dispatchServiceForm;
      const isValid = formRoot?.isValid();
      const { username, id } = formRoot?.getJsonValue();
      if (!isValid) {
        onComplete && onComplete({ success: false });
        return false;
      }

      const { success, msg } = id
        ? await updateBusinessContact({
            mask: false,
            data: {
              username,
              id,
            },
            hideErr: true,
          })
        : await deleteBusinessContact({
            mask: false,
            hideErr: true,
          });

      if (!success) {
        error({
          content: msg,
        });
        onComplete && onComplete({ success: false });
        return false;
      }
      successMessage($i('partner.production.module.modify_success'));
      onComplete && onComplete({ success: true });
      dispatchServiceWin.close();
    },
    onCancel: () => {
      onComplete && onComplete({ success: false, isCancel: true });
    },
  });
}
