# 导出服务清单
1. index.ts 文件

   
这个文件主要负责导出服务清单功能的核心逻辑，包括打开导出服务的弹窗、表单验证、数据处理和文件下载

```js
import { useModal, useMessage } from '@xcentral/hooks/use_prompt';
import { ref, h } from '@vue/composition-api';

import ExportForm from './form.vue';
import { $i } from '@/framework/i18n';
import { Params as CustomerFilterParams } from '@/api/customer_operation/types/get_customer_names';
import { getCustomerNames } from '@/api/customer_operation';
import { downloadFile } from '@/util/download_file';

/** 任务进行中 */
const errTasking = 1027;
/** 导出为空 */
const errExportEmpty = 1028;

export function openExportService(data: {
  params: CustomerFilterParams;
  isDetail?: boolean;
  serviceIds?: string[];
}) {
  const contentRef = ref();
  contentRef.value = h(ExportForm, {
    props: {
      params: data.params,
      isDetail: data.isDetail,
    },
    ref: 'exportFormRef',
  });
  useModal().open({
    width: 480,
    header: $i('partner.customer_operation.export_service'),
    destroyOnHide: true,
    content: contentRef.value,
    onOk: async () => {
      const formRoot = contentRef.value.context.$refs.exportFormRef;
      const isValid = formRoot?.isValid();
      if (!isValid) {
        return false;
      }
      const value = formRoot.getJsonValue();
      let isSuccess = true;

      if (data.params.is_all) {
        const { success, data: customerNames } = await getCustomerNames({
          data: data.params,
        });
        if (!success) {
          return false;
        }
        data.params.corp_names = customerNames?.customer_names ?? [];
      }

      await downloadFile('/scl/v1/partner/customer/prosecution/export/list', {
        method: 'post',
        params: {
          service_ids: data.serviceIds ?? [],
          corp_names: data.params.corp_names,
          ...value,
        },
        cb: ({ code, msg }) => {
          isSuccess = false;
          if ([errTasking, errExportEmpty].includes(code)) {
            useModal().error({
              content: msg,
            });
          } else {
            useModal().error({
              title: $i('partner.customer_operation.export_err_title'),
              content: $i('partner.customer_operation.export_err_content', {
                phone: window.BBC.i18nConfig.AFTER_SALE_PHONE,
              }),
            });
          }
        },
      });
      isSuccess && useMessage().success($i('common.operation.success_message'));
      return isSuccess;
    },
  });
}
```


主要逻辑：

打开导出服务弹窗：使用 useModal().open 方法打开一个包含导出表单的弹窗。

表单验证：在点击确认按钮时，调用表单的 isValid 方法进行验证。

数据处理：如果 params.is_all 为真，则调用 getCustomerNames 获取所有客户名称。

文件下载：调用 downloadFile 方法下载文件，并处理下载结果的回调。

3. form.vue 文件

这个文件主要负责导出服务清单的表单界面和表单验证逻辑。
```js
<template>
  <div class="export-service-form">
    <IxAlert type="offline" class="export-service-form__alert">
      <div v-if="!isDetail">
        <i18nIntl path="partner.customer_operation.export_service.alert">
          <span slot="total">{{ total }}</span>
        </i18nIntl>
      </div>
      <div v-else>
        {{ $i('partner.customer_operation.export_service.detail_alert') }}
      </div>
    </IxAlert>
    <IxForm labelAlign="start" :control="formGroup" :labelCol="6" colonless>
      <IxFormItem
        :label="$i('partner.customer_operation.select_service')"
        required
      >
        <SelectService
          size="md"
          control="serviceType"
          :maxLabel="1"
          @hasFetch="initSelectService"
        />
      </IxFormItem>
      <IxFormItem
        :label="$i('partner.customer_operation.renewal_follow_up_status')"
        required
      >
        <RenewalFollowStatusComp
          size="md"
          control="followStatus"
          :maxLabel="1"
          :hiddenStatus="[RenewalFollowStatus.reordered]"
          :all.sync="allFollowStatus"
        />
      </IxFormItem>
      <IxFormItem :label="$i('partner.customer_operation.export_service.time')">
        <DateRangePicker control="dateRange" />
      </IxFormItem>
    </IxForm>
  </div>
</template>

<script lang="ts" setup>
import { ref } from '@vue/composition-api';
import { useFormGroup } from '@idux-vue2/cdk/forms';
import RenewalFollowStatusComp from '../renewal_follow_status.vue';
import SelectService from '../select_service.vue';
import DateRangePicker from '../date_range_picker.vue';
import { useCustomerNames } from '../../hooks/use_customer_names';
import { Params as CustomerFilterParams } from '@/api/customer_operation/types/get_customer_names';
import { Validators } from '@idux-vue2/cdk/forms';
import { startOfYear, endOfYear, addYears, isAfter } from 'date-fns';
import { $i } from '@/framework/i18n';
import { RenewalFollowStatus } from '../../const';

const props = defineProps<{
  params: CustomerFilterParams;
  isDetail: boolean;
}>();

const allFollowStatus = ref(false);
const { total } = useCustomerNames(props.params);
const now = new Date();
const formGroup = useFormGroup<{
  serviceType?: number[];
  followStatus?: number[];
  dateRange: [Date, Date];
}>({
  serviceType: [undefined, Validators.required],
  followStatus: [[RenewalFollowStatus.wait], Validators.required],
  dateRange: [
    [startOfYear(now), endOfYear(now)],
    [
      Validators.required,
      (value: [Date, Date] | undefined) => {
        if (value?.[0] && value?.[1]) {
          let start = addYears(value[0], 1);
          const isError = isAfter(value[1], start);

          if (isError) {
            return {
              confirm: {
                message: $i('partner.customer_operation.date_range_picker.err'),
              },
            };
          }
        }
      },
    ],
  ],
});
const initSelectService = (data: number[]) => {
  formGroup.setValue({ serviceType: data });
};
const getJsonValue = () => {
  const value = formGroup.getValue();

  return {
    service_type: value.serviceType,
    follow_status: allFollowStatus.value ? [] : value.followStatus,
    start_time: Math.floor(value.dateRange[0].getTime() / 1000),
    end_time: Math.floor(value.dateRange[1].getTime() / 1000),
  };
};

const isValid = () => {
  if (formGroup.valid.value) {
    return true;
  } else {
    formGroup.markAsDirty();
    return false;
  }
};

defineExpose({
  getJsonValue,
  isValid,
});
</script>

  }
}
</style>
```

主要逻辑：

表单组件：使用 IxForm 组件构建表单，包含选择服务类型、续费跟进状态和时间范围的输入项。

表单验证：使用 useFormGroup 创建表单组，并定义各个字段的验证规则
。

数据处理：提供 getJsonValue 方法将表单数据转换为 JSON 格式，供外部调用。

表单有效性检查：提供 isValid 方法检查表单是否有效。

总结

导出服务清单功能通过三个文件协同工作：

index.ts 负责核心逻辑，包括打开弹窗、表单验证和文件下载。

table.vue 提供导出按钮和参数计算。

form.vue 构建表单界面和表单验证逻辑。

这样设计的好处是职责分明，易于维护和扩展。
