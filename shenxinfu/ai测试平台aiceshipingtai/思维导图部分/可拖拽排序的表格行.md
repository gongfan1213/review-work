可拖拽排序的表格行功能。这个方法使用了 Vue.js 框架，并且依赖于 Sortable 库来实现拖拽排序的效果。下面我将逐行解释这段代码的功能：


```js
treeRowDrop () {
  // 在DOM更新后执行回调函数
  this.$nextTick(() => {
    // 获取到组件中的tree引用
    const xTable = this.$refs.tree
    // 通过查询选择器获取到表格的tbody元素
    const tbody = xTable.$el.querySelector('.body--wrapper>.vxe-table--body tbody')
    // 使用Sortable库创建一个新的Sortable实例，使tbody中的行可拖拽排序
    this.sortableRowData = Sortable.create(tbody, {
      // 指定拖拽把手的选择器
      handle: '.vxe-body--row',
      // 允许排序
      sort: true,
      // 当拖拽开始时设置传输的数据
           // 当拖拽开始时设置传输的数据
      setData: (dataTransfer, dragEl) => {
        // 获取被拖拽的行对应的数据
        const selfRow = xTable.getRowNode(dragEl).item
        // 获取选中的行数据
        const selectionList = this.getChecked().concat(this.getReserveChecked())
        // 从选中的行数据中提取id
        const issueIds = selectionList.map(i => i.id)
        // 如果允许多行拖拽且选中的行中包含被拖拽的行，则设置数据为选中的行
        if (this.allowMultiDrag && selectionList.length && issueIds.includes(selfRow.id)) {
          dataTransfer.setData('data', JSON.stringify(selectionList))
        } else {
          // 否则只设置被拖拽的行的数据
          dataTransfer.setData('data', JSON.stringify(selfRow))
        }
      },
      // 当拖拽移动时的回调函数
         // 当拖拽移动时的回调函数
      onMove: evt => {
        // 触发自定义事件sortableMove
        this.$emit('sortableMove', evt)
        // 清除所有行的背景色
        const rows = evt.target.rows
        for (var i = 0; i < rows.length; i++) {
          rows[i].style.backgroundColor = ''
        }
        // 设置当前拖拽到的行的背景色
        evt.related.style.backgroundColor = '#dfecfb'
        // 返回false禁用排序，这里可能是因为有特殊的父子级功能需要处理
        return false
      },
      // 当拖拽结束时的回调函数
      onEnd: evt => {
        // 获取被拖拽的行对应的数据
        const { item } = evt
        this.selfRow = xTable.getRowNode(item).item
        // 通过原生事件获取目标节点
        const targetTrElem = this.getTargetElem(evt.originalEvent.srcElement)
        if (targetTrElem) {
          const prevTrElem = targetTrElem.parentNode
          if (prevTrElem) {
            // 清除目标节点的背景色
            prevTrElem.style.backgroundColor = ''
            // 获取目标节点的数据
            this.prevRow = xTable.getRowNode(prevTrElem).item
          } else {
            // 如果没有目标节点，则设置为顶级节点
            this.prevRow = { id: '' }
          }
        } else {
          this.prevRow = null
        }
        
          // 清除所有行的背景色
        const rows = evt.target.rows
        for (var i = 0; i < rows.length; i++) {
          rows[i].style.backgroundColor = ''
        }
        // 获取选中的行数据
        const selectionList = this.getChecked().concat(this.getReserveChecked())
        const issueIds = selectionList.map(i => i.id)
        let sourceData
        // 如果允许多行拖拽且选中的行中包含被拖拽的行，则sourceData为选中的行
        if (this.allowMultiDrag && selectionList.length && issueIds.includes(this.selfRow.id)) {
          sourceData = selectionList
        }
        // 触发自定义事件sortableEnd
        this.$emit('sortableEnd', evt, sourceData || this.selfRow, this.prevRow, (node, loading) => {
          // 重新加载行数据
          node && this.reloadRow([node])
          // 设置加载状态
          this.loading = loading
        })
      }
    })
  })
},
```
