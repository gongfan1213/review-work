<template>
  <div class="design-table">
    <TableToolbar class="table-toolbar" :productId="productId" :projectId="projectId" @reload="reload" @submit="submit">
    </TableToolbar>
    <IxTable
      class="table-content"
      :columns="columns"
      :dataSource="data"
      ellipsis
      autoHeight
      :pagination="pagination">
      <template #title="{ record }">
        <span class="column-name" @click="designFunc(record)">{{ record.name }}</span>
      </template>
      <template #ipd_project_name="{ record }">
        <span>{{ record.ipd_project_name || '-' }}</span>
      </template>
      <template #related_issues="{ record }">
        <el-popover placement="bottom" trigger="click">
          <IxTable
            :columns="relatedIssueColumns"
            :dataSource="relatedIssues"
            :pagination="false"
            empty="simple"
            getKey="id"
            size="sm"
            ellipsis
            :borderless="false"
            :scroll="{ height: 200 }"
            style="width: 500px"
          >
            <template #issueName="{ record }">
              <div class="issue-node" style="cursor: pointer" @click="showIssueDetail(record)">
                <span class="issue-category" :class="record.issue_category.key"></span>
                <span class="issue-name" :title="record.name">{{ record.id + '-' + record.name }}</span>
              </div>
            </template>
          </IxTable>
          <span
            slot="reference"
            style="color: #1c6eff; cursor: pointer; display: flex; width: 100%"
            @click="() => showIssueFunc(record)"
          >{{ record.issue_id ? record.issue_id.split(',').length : 0 }}</span
          >
        </el-popover>
      </template>
      <template #created_at="{ record }">
        {{ formatDate(record.created_at, true) }}
      </template>
      <template #task_step_display="{ record }">
        <span>{{ record.task_step_display || '-' }}</span>
      </template>
      <template #task_step_state_display="{ record }">
        <span style="width: 72px;" title="">
          <img
            v-if="record.task_step_state_display"
            :src="getImageSrc(record.task_step_state_display)"
            style="width: 16px;"
          />
          {{ record.task_step_state_display || '-' }}
          <IxTooltip v-if="record.task_step_state_display === '失败'" :title="record.state_message">
            <img src="./imgs/question.svg" alt="question" style="margin-left: 2px;width: 16px;" />
          </IxTooltip>
        </span>
      </template>

      <template #update_at="{ record }">
        {{ formatDate(record.update_at, true) }}
      </template>
      <template #action="{ record }" style="white-space: nowrap;">
        <IxButtonGroup :gap="20" mode="link" class="button-group" style="white-space: nowrap;">
          <IxButton @click="editFunc(record)">编辑</IxButton>
          <IxButton danger @click="deleteFunc(record)">删除</IxButton>
        </IxButtonGroup>
      </template>
    </IxTable>
    <!-- 编辑 -->
    <AddModal :visible.sync="editVisible" :productId="productId" :record="handleRecord" @submit="submit" />
    <!-- 删除 -->
    <IxModal
      :visible.sync="confirmDeleteVisible"
      type="confirm"
      header="删除"
      :title="`确定删除测试设计【${handleRecord.name}】？`"
      :onOk="confirmDelete"
      destroyOnHide
    >
      <p style="color: #6f7785">删除后数据不可恢复，请谨慎操作！</p>
    </IxModal>
  </div>
</template>
<script>
import { formatDate } from '@/utils/date'
import { getDesignList, createDesign, deleteDesign, updateDesign } from '@/api/test_design'
import { getIssuesByIds } from '@/api/ipd'
import TableToolbar from './TableToolbar.vue'
import AddModal from './modal/AddModal.vue'
import { columns, relatedIssueColumns } from './tableColumn'
import { mapGetters } from 'vuex'

export default {
  components: {
    TableToolbar,
    AddModal
  },
  props: {
    productId: {
      type: [Number, null],
      default: null
    },
    projectId: {
      type: [Number, null],
      default: null
    }
  },
  data () {
    return {
      columns,
      relatedIssueColumns,
      data: [],
      searchData: {
        name: null,
        create_by: null,
        update_by: null,
        issue_id: [],
        task_step: null,
        task_step_state: null

      },
      pagination: {
        total: 0,
        pageIndex: 1,
        pageSize: 20,
        showTotal: true,
        showSizeChanger: true,
        onChange: (pageIndex, pageSize) => {
          if (this.pagination.pageSize !== pageSize) {
            this.pagination.pageIndex = 1
            this.pagination.pageSize = pageSize
            this.getDesignList()
          } else {
            this.pagination.pageIndex = pageIndex
            this.getDesignList()
          }
        }
      },
      confirmDeleteVisible: false,
      editVisible: false,
      relateVisible: false,
      handleRecord: {},
      showIssue: false,
      relatedIssues: [],
      reloadTimer: null,
      timerDelay: 5
    }
  },
  computed: {
    ...mapGetters(['globalProject'])
  },
  watch: {
    projectId () {
      this.getDesignList()
    }
  },
  created () {
    this.getDesignList()
  },
  // 销毁前清空定时器
  beforeDestroy () {
    if (this.reloadTimer) {
      clearTimeout(this.reloadTimer)
      this.reloadTimer = null
    }
  },
 methods: {
    formatDate,
    reload (searchData) {
      this.pagination.pageIndex = 1
      this.searchData = searchData
      this.getDesignList()
    },
    setTimer () {
      if (this._isDestroyed) return
      if (this.reloadTimer) {
        clearTimeout(this.reloadTimer)
      }
      // 判断浏览器当前标签页是否激活 激活的话5s后刷新表格，没激活等待5s后继续判断页面是否激活
      this.reloadTimer = setTimeout(() => {
        if (!document.hidden) {
          this.getDesignList()
        } else {
          this.setTimer()
        }
      }, this.timerDelay * 1000)
    },
    getImageSrc (state) {
      const stateToImageMap = {
        '排队中': require('@/views/testdesign/imgs/PENDING.svg'),
        '进行中': require('@/views/testdesign/imgs/STARTED.svg'),
        '成功': require('@/views/testdesign/imgs/SUCCESS.svg'),
        '已中止': require('@/views/testdesign/imgs/REVOKED.svg'),
        '失败': require('@/views/testdesign/imgs/FAILURE.svg')
      }
      return stateToImageMap[state] || ''
    },
    getDesignList () {
      if (!this.projectId) {
        this.data = []
        this.pagination.total = 0
        return
      }
      getDesignList({
        paginate: true,
        belong_project_id: this.projectId,
        page: this.pagination.pageIndex,
        per: this.pagination.pageSize,
        name: this.searchData.name,
        issue_id: this.searchData.issue_id.join() || null,
        create_by:
          this.searchData.create_by && this.searchData.create_by.length ? this.searchData.create_by.join() : null,
        update_by:
          this.searchData.update_by && this.searchData.update_by.length ? this.searchData.update_by.join() : null,
        task_step:
          this.searchData.task_step && this.searchData.task_step.length ? this.searchData.task_step.join() : null,
        task_step_state:
          this.searchData.task_step_state && this.searchData.task_step_state.length ? this.searchData.task_step_state.join() : null
      }).then((resp) => {
        resp.data.map((item) => {
          item.key = item.id
          item.showIssue = false
        })
        this.data = resp.data
        this.pagination.total = resp.total
        this.setTimer()
      })
    },
    showIssueFunc (record) {
      this.data.map((item) => {
        item.showIssue = false
      })
      this.relatedIssues = []
      record.showIssue = true
      this.getIssueList(record)
    },
    getIssueList (record) {
      if (!record.issue_id) return
      getIssuesByIds({
        issue_ids: record.issue_id
      }).then((res) => {
        this.relatedIssues = res.data
      })
    },

submit (formData, isEdit) {
      if (this.loading) return
      this.loading = true
      if (isEdit) {
        updateDesign(this.handleRecord.id, {
          name: formData.name,
          issue_id: formData.issue_id,
          belong_project_id: this.projectId
        }).then((resp) => {
          if (resp && resp.id) {
            this.$message.success('更新成功')
            this.getDesignList()
          }
          this.loading = false
        })
      } else {
        createDesign({
          name: formData.name,
          issue_id: formData.issue_id,
          belong_project_id: this.projectId
        }).then((resp) => {
          if (resp && resp.id) {
            const { product_id, project_id } = this.globalProject
            this.$router.push({
              path: `/DesignHome/AddDesign/${resp.id}`,
              query: { name: resp.name, productId: product_id, projectId: project_id, generate: true }
            })
            this.getDesignList()
          }
          this.loading = false
        })
      }
    },
    designFunc (record) {
      const { product_id, project_id } = this.globalProject
      this.$router.push({
        path: `/DesignHome/AddDesign/${record.id}`,
        query: { name: record.name, productId: product_id, projectId: project_id }
      })
    },
    editFunc (record) {
      this.handleRecord = record
      this.editVisible = true
    },
    relatedFunc (record) {
      this.handleRecord = record
      this.relateVisible = true
    },
    deleteFunc (record) {
      this.handleRecord = record
      this.confirmDeleteVisible = true
    },
    confirmDelete () {
      deleteDesign(this.handleRecord.id, this.globalProject.product_id).then(() => {
        this.$message.success('删除成功')
        this.getDesignList()
      })
    },
    showIssueDetail (record) {
      window.open(`https://ipd.atrust.sangfor.com/ipd/product/${record.product.id}/issue/${record.id}`, '_blank')
    }
  }
}

<template>
  <el-dialog
    class="add-modal"
    :title="`${record ? '编辑' : '新增'}测试设计`"
    width="1050px"
    :visible.sync="addVisible"
    :closeOnClickModal="false"
    @open="initFormData"
    @close="closeModal">
    <el-form
      :model="formData"
      size="small"
      :rules="rules"
      ref="formData"
      class="add-form"
      v-if="addVisible">
      <el-form-item label="测试设计名称" prop="name" class="demand-name">
        <el-input v-model="formData.name" placeholder="请输入测试设计名称"></el-input>
      </el-form-item>
      <el-form-item label="关联需求" prop="issue_id" class="is-required">
        <Associate ref="associate" class="add-content" :productId="productId" :projectId="projectId" :issueId="formData.issue_id"></Associate>
      </el-form-item>
    </el-form>
    <div slot="footer" class="dialog-footer">
      <el-button size="small" type="primary" @click="handleOk">确 定</el-button>
      <el-button size="small" @click="closeModal">取 消</el-button>
    </div>
  </el-dialog>
</template>
export default {
  props: {
    visible: {
      type: Boolean,
      default: false
    },
    record: {
      type: Object,
      default: () => {}
    },
    productId: {
      type: [Number, null],
      default: null
    },
    projectId: {
      type: [Number, null],
      default: null
    }
  },
  components: {
    Associate
  },
  data () {
    return {
      addVisible: false,
      isEdit: false,
      formData: {
        name: '',
        issue_id: []
      },
      rules: {
        name: [
          { required: true, message: '测试设计名称不支持为空', trigger: 'blur' },
          { max: 250, message: '最大字符255', trigger: 'blur' }
        ]
      }
    }
  },
 watch: {
    visible (val) {
      this.addVisible = val
    }
  },
  methods: {
    initFormData () {
      this.isEdit = true
      this.formData = {
        name: this.record ? this.record.name : '',
        issue_id: this.record ? this.record.issue_id.split(',') : []
      }
    },
    handleOk () {
      this.$refs.formData.validate((valid) => {
        if (valid) {
          if (!this.$refs.associate.selectionList.length) {
            Message.error('请选择关联需求')
            return
          }
          this.formData.issue_id = this.$refs.associate.selectionList.map(item => item.id).join()
          console.log(this.formData)
          this.$emit('submit', this.formData, this.isEdit)
          this.closeModal()
        } else {
          return false
        }
      })
    },
    closeModal () {
      this.formData = {
        name: '',
        issue_id: []
      }
      this.$refs.formData.clearValidate()
      this.$emit('update:visible', false)
    }
  }
}
