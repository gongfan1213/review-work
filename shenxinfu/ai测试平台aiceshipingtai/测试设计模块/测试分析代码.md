```js
  <IxModal
      :visible.sync="archiveVisible"
      type="confirm"
      header="提示"
      title="将根据测试分析的结果自动生成测试点，生成将覆盖现有内容。是否继续？"
    >
      <template #footer>
        <IxButton mode="primary" @click="startAiCaseDesign()">确认</IxButton>
        <IxButton @click="archiveVisible = false">取消</IxButton>
      </template>
    </IxModal>

    <Mindmap ref="mindmap" :isDisabled="isDisabled" :isCurrentTab="isCurrentTab" currentTab="TEST_ANALYSIS"></Mindmap>
  </div>
</template>
export default {
  name: 'DesignAnalysisView',
  components: {
    Mindmap,
    DesignHeader
  },

  props: {
    demandtotal: {
      type: Number,
      default: 0
    },
    readonly: {
      type: Boolean,
      default: false
    },
    designId: {
      type: [Number, String],
      default: null
    },
    saveData: {
      type: Function,
      default: () => {}
    },
    checkTestFactorChange: {
      type: Function,
      default: () => {}
    },
    checkTestCaseChange: {
      type: Function,
      default: () => {}
    }
  },
  data () {
    return {
      topbarVisible: false,
      searchValue: '',
      visible: false,
      confirmVisble: false,
      addVisible: false,
      confirmVisible: false,
      isStopVisible: false,
      loading: false,
      archiveVisible: false
    }
  },
 computed: {
    ...mapState({
      testCaseState: (state) => state.mindmap.testCaseState
    }),
    isCurrentTab () {
      return this.testCaseState.task_step_display === '测试分析'
    },
    isDisabled () {
      return ['排队中', '进行中'].includes(this.testCaseState.task_step_state_display)
    }
  },
  methods: {
    ...mapMutations(['setTestCaseState']),
    toggleConfirmModal () {
      this.confirmVisible = !this.confirmVisible
    },
    openAddDemandModal () {
      this.confirmVisible = false
      this.addVisible = true
    },
    handleGenerateTestPoints () {
      this.confirmVisible = true
    },
    async handleGenerateTestCasePoints () {
      this.archiveVisible = true
    },
    stopAiDesign () {
      try {
        this.$refs.mindmap && this.$refs.mindmap.stopAiDesign()
      } catch (error) {
        this.$message.error('停止全量生成测试项失败')
        console.error(error)
      } finally {
      }
    },
 async startAiDesign (isPartDesign = false, { action } = { action: 'TEST_ANALYSIS' }) {
      try {
        if ((isPartDesign && this.checkTestFactorChange()) || (!this.isCheckedTestTab && this.checkTestCaseChange())) {
          await this.saveData(true)
        }
        this.$refs.mindmap && this.$refs.mindmap.aiDesign(isPartDesign, { action: 'TEST_ANALYSIS' })
        this.loading = true
      } finally {
        this.confirmVisible = false
        this.loading = false
      }
    },
    async startAiCaseDesign (isPartDesign = false, { action } = { action: 'TEST_DESIGN' }) {
      this.$emit('update-active-tab', '3')
      try {
        if ((isPartDesign && this.checkTestFactorChange()) || (!this.isCheckedTestTab && this.checkTestCaseChange())) {
          await this.saveData(true)
        }
        setTimeout(() => {
          this.$bus.$emit('aiDesign', false, { action: 'TEST_DESIGN' })
        }, 300)
      } finally {
        this.confirmVisible = false
        this.loading = false
        this.archiveVisible = false
      }
    }
  }

}
```
