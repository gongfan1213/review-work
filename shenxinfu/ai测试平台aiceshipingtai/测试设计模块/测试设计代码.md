```js
<IxModal
      :visible.sync="archiveVisible"
      type="confirm"
      header="提示"
      title="功能正在开发中..."
    >
      <p style="color: #6f7785">
        建议使用导出功能，本地转化格式后再上传平台归档用例。
      </p>
      <template #footer>
        <IxButton @click="archiveVisible = false">我知道了</IxButton>
      </template>
    </IxModal>

    <Mindmap ref="mindmap" :isDisabled="isDisabled" :isCurrentTab="isCurrentTab" currentTab="TEST_DESIGN"></Mindmap>
  </div>
</template>
export default {
  name: 'TestPlanView',
  components: {
    Mindmap,
    DesignHeader
  },
  props: {
    demandtotal: {
      type: Number,
      default: 0
    },
    designId: {
      type: [Number, String],
      default: null
    },
    saveData: {
      type: Function,
      default: () => {}
    },
    checkTestFactorChange: {
      type: Function,
      default: () => {}
    },
    checkTestCaseChange: {
      type: Function,
      default: () => {}
    }
  },

  data () {
    return {
      searchValue: '',
      visible: false,
      confirmVisible: false,
      addVisible: false,
      archiveVisible: false // 控制归档用例弹窗的显
    }
  },
  computed: {
    ...mapState({
      testCaseState: (state) => state.mindmap.testCaseState
    }),
    isCurrentTab () {
      return this.testCaseState.task_step_display === '测试设计'
    },
    isDisabled () {
      return ['排队中', '进行中'].includes(this.testCaseState.task_step_state_display)
    }
  },

  methods: {
    toggleConfirmModal () {
      this.confirmVisible = !this.confirmVisible
    },
    openAddDemandModal () {
      this.addVisible = true
    },
    openArchiveModal () {
      this.archiveVisible = true
    },
    async handleGenerateTestPoints () {
      this.confirmVisible = true
    },
    stopAiDesign () {
      try {
        this.$refs.mindmap && this.$refs.mindmap.stopAiDesign()
      } catch (error) {
        this.$message.error('停止全量生成测试项失败')
        console.error(error)
      } finally {
      }
    },
    async startAiDesign (isPartDesign = false, { action } = { action: 'TEST_DESIGN' }) {
      try {
        if ((isPartDesign && this.checkTestFactorChange()) || (!this.isCheckedTestTab && this.checkTestCaseChange())) {
          await this.saveData(true)
        }
        this.$refs.mindmap && this.$refs.mindmap.aiDesign(isPartDesign, { action: 'TEST_DESIGN' })
      } finally {
        this.confirmVisible = false
        this.loading = false
      }
    }
  }
}
```
