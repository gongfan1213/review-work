```js
<template>
  <div class="design-table" :class="{ processVisible: processVisible }">
    <!-- 设计头部组件 -->
    <DesignHeader
      title="测试需求"
      :isCurrentTab="isCurrentTab"
      :isDisabled="isDisabled"
      :demandtotal="demandtotal"
      :confirmVisible.sync="confirmVisible"
      :addVisible.sync="addVisible"
      primaryActionText="生成需求项"
      secondaryActionText="【AI生成测试分析】"
      confirmTitle="将根据关联需求生成一份测试需求内容，生成将覆盖现有内容。是否继续？"
      infoText="内容核对完成后，建议执行下一步"
      disabledTooltip="AI任务正在进行中，请等待结果出来后再操作"
      @open-add-demand-modal="openAddDemandModal"
      @primary-action="handleGenerateDemand"
      @secondary-action="handleAIAnalysis"
      @stop-action="handleStopFullDesign"
      @confirm-action="handleStartFullDesign"
      @search="handleSearch"
    />
    <!-- AI确认模态框 -->
    <IxModal
      ref="aiConfirmModal"
      :visible.sync="confirmAIVisible"
      type="confirm"
      title="将根据需求项生成测试分析脑图，生成将覆盖已有测试分析脑图。是否继续？"
    >
      <template #footer>
        <IxButton mode="primary" @click="startAiCaseAnalysis()">确认</IxButton>
        <IxButton @click="confirmAIVisible = false">取消</IxButton>
      </template>
    </IxModal>
    <div class="demand-table">
      <IxProTable
        :autoHeight="!isLoading"
        class="table-content"
        :columns="columns"
        :dataSource="!isCurrentTab || (isCurrentTab && !isDisabled) ? data : []"
        :ellipsis="isLoading"
        :pagination="pagination"
        getKey="id"
      >
        <template #description="{ value }">
          <p v-html="value.replace(/\n/g, '<br>')"></p>
        </template>
        <template v-if="isLoading" #empty>
          <div></div>
        </template>
        <template v-else #empty>
          <div class="empty">暂无数据</div>
        </template>
        <template #operation="{ record }">
          <IxButtonGroup :gap="10" mode="link" class="button">
            <IxButton @click="editRecord(record)" class="button-group" :disabled="isDisabled">编辑</IxButton>
            <IxButton
              @click="deleteRecord(record)"
              danger
              class="button-group"
              color="red"
              :disabled="isDisabled"
            >删除</IxButton>
          </IxButtonGroup>
        </template>
      </IxProTable>
      <!-- 加载中显示 -->
      <div class="loading" v-if="isLoading">
        <div class="rotated-image">
          <img src="../imgs/loading.svg" />
        </div>
        <div class="analysis-text">需求分析中</div>
        <div class="instruction-text">
          为确保AI能准确生成测试设计，请核对并调整分析结果（需求项），通过增删改以提升其清晰度和准确性
          离开本页不会影响需求分析结果，后台会自动处理并生成结果。
        </div>
      </div>
      <!-- 删除确认模态框 -->
      <IxModal :visible.sync="confirmDeleteVisible" type="confirm" title="确定删除需求项吗？">
        <p style="color: #6f7785">删除后数据不可恢复，请谨慎操作！</p>
        <template #footer>
          <IxButton mode="primary" @click="confirmDelete">确认</IxButton>
          <IxButton @click="confirmDeleteVisible = false">取消</IxButton>
        </template>
      </IxModal>
      <!-- 添加需求项组件 -->
      <AddDemandItem :visible.sync="addVisible" :record="handleRecord" @close="handleClose" @submit="handleAddDemand" />
    </div>
  </div>
</template>
export default {
  name: 'TestDemandView',
  components: {
    AddDemandItem,
    DesignHeader
  },
  props: {
    designId: {
      type: [Number, String],
      default: null
    }
  },
  data () {
    return {
      data: [], // 需求项数据
      columns: [
        {
          title: '序号',
          type: 'indexable',
          dataKey: 'index',
          resizable: true,
          width: 50
        },
        {
          title: '测试需求项',
          dataKey: 'demand_name',
          customCell: 'demand_name',
          resizable: true,
          width: 400
        },
        {
          title: '需求描述',
          dataKey: 'description',
          customCell: 'description',
          resizable: true
        },
        {
          title: '操作',
          key: 'operation',
          customCell: 'operation',
          resizable: true,
          width: 260
        }
      ],
      pagination: {
        total: 0,
        pageIndex: 1,
        pageSize: 10,
        showTotal: true,
        showSizeChanger: true,
        onChange: (pageIndex, pageSize) => {
          if (this.pagination.pageSize !== pageSize) {
            this.pagination.pageIndex = 1
            this.pagination.pageSize = pageSize
            this.getDetailList()
          } else {
            this.pagination.pageIndex = pageIndex
            this.getDetailList()
          }
        }
      },
      demandtotal: 0, // 需求总数
      searchValue: '', // 搜索值
      searchvisible: false, // 搜索框是否可见
      confirmVisible: false, // 确认框是否可见
      addVisible: false, // 添加需求项模态框是否可见
      confirmAIVisible: false, // AI确认模态框是否可见
      confirmDeleteVisible: false, // 删除确认模态框是否可见
      editVisible: false, // 编辑模态框是否可见
      handleRecord: {} // 当前操作的记录
    }
  },
  computed: {
    ...mapState({
      testCaseState: (state) => state.mindmap.testCaseState,
      processVisible: (state) => state.mindmap.processVisible
    }),
    isCurrentTab () {
      return this.testCaseState.task_step_display === '需求分析'
    },
    isDisabled () {
      return ['排队中', '进行中'].includes(this.testCaseState.task_step_state_display)
    },
    isLoading () {
      return this.isCurrentTab && this.isDisabled
    }
  },
  methods: {
    ...mapMutations(['setTestCaseState', 'setTimer']),
    // 获取需求项列表
    async getDetailList () {
      try {
        const params = {
          action: 'DEMAND_ANALYSIS',
          per: this.pagination.pageSize,
          page: this.pagination.pageIndex
        }
        const response = await getAiDesignState(this.designId, params)
        if (response && response.data) {
          this.setTestCaseState(response.data)
          this.data = response.data.data.sort((a, b) => new Date(b.created_at) - new Date(a.created_at))
          this.pagination.total = response.total
          this.demandtotal = this.pagination.total
          this.$emit('update-demandtotal', this.demandtotal)
          this.$nextTick(() => {
            if (
              response.data.task_step === 'DEMAND_ANALYSIS' &&
              ['排队中', '进行中'].includes(response.data.task_step_state_display)
            ) {
              this.setTimer(() => {
                this.getDetailList()
              })
            }
          })
        }
      } catch (error) {
        this.$message.error('获取需求项列表失败')
      }
    },
    // 搜索需求项
    // 搜索需求项
    async handleSearch (searchValue) {
      this.searchValue = searchValue
      const params = {
        page: this.pagination.page,
        per: this.pagination.per,
        demand_name: this.searchValue,
        test_design_id: this.designId
      }
      const response = await searchDemandMatrix(params)
      if (response) {
        this.data = response.data
        this.pagination.total = response.data.length
      }
    },
    // 关闭添加需求项模态框
    handleClose () {
      this.addVisible = false
      this.$refs.addDemandItem.closeModal()
    },
    // 添加或更新需求项
    async handleAddDemand (formData) {
      try {
        if (this.handleRecord.id) {
          await updateTestDemand(this.handleRecord.id, {
            demand_name: formData.demand_name,
            description: formData.description
          })
        } else {
          await addTestDemand({
            test_design_id: this.designId,
            demand_name: formData.demand_name,
            description: formData.description
          })
        }
        this.$message.success('操作成功')
        this.getDetailList()
      } catch (error) {
        console.error(error)
      }
    },

 // 删除需求项
    async handleDelete (record) {
      try {
        const response = await deleteTestDemand(record.id)
        if (response) {
          await this.getDetailList()
          this.$message.success('删除需求项成功')
        }
      } catch (error) {
        this.$message.error('删除需求项失败')
      }
    },
    // 开始全量设计
    async handleStartFullDesign () {
      this.confirmVisible = false
      this.confirmAIVisible = false
      const response = await aiDesign(this.designId, { action: 'DEMAND_ANALYSIS' })
      if (response && response.data) {
        this.setTestCaseState(response.data)
        this.data = response.data.data
        this.$message.success('操作成功')
        this.getDetailList()
      }
    },
    // 停止全量设计
    async handleStopFullDesign () {
      const response = await stopAiDesign(this.designId, { action: 'DEMAND_ANALYSIS' })
      if (response && response.data) {
        this.setTestCaseState(response.data)
        this.data = response.data.data
        this.$message.success('暂停全量设计成功')
      }
    },
    // 处理AI分析
    handleAIAnalysis () {
      if (this.demandtotal === 0) {
        this.$message.error('需求项列表为空，不符合AI分析要求')
      } else {
        this.toggleAIConfirmModal()
      }
    },
 // 开始AI案例分析
    startAiCaseAnalysis () {
      this.confirmVisible = false
      this.confirmAIVisible = false
      this.$emit('update-active-tab', '2')
      setTimeout(() => {
        this.$bus.$emit('aiDesign', false, { action: 'TEST_ANALYSIS' })
      }, 300)
    },
    // 取消搜索
    cancelSearch () {
      this.searchValue = ''
    },
    // 切换确认模态框
    toggleConfirmModal () {
      this.confirmVisible = !this.confirmVisible
    },
    // 切换AI确认模态框
    toggleAIConfirmModal () {
      this.confirmAIVisible = !this.confirmAIVisible
    },
    // 打开添加需求项模态框
    openAddDemandModal () {
      this.handleRecord = {}
      this.addVisible = true
    },
    // 编辑记录
    editRecord (record) {
      this.handleRecord = { ...record }
      this.addVisible = true
    },
    // 删除记录
    deleteRecord (record) {
      this.handleRecord = record
      this.confirmDeleteVisible = true
    },
    // 确认删除
   confirmDelete () {
      this.handleDelete(this.handleRecord)
      this.confirmDeleteVisible = false
    },
    // 生成需求项
    handleGenerateDemand () {
      if (this.demandtotal > 0) {
        this.toggleConfirmModal()
      } else {
        this.handleStartFullDesign()
      }
    }
  },
  mounted () {
    const generate = this.$route.query.generate
    if (generate) {
      this.handleStartFullDesign()
      this.$router.replace({
        ...this.$route,
        query: {
          ...this.$route.query,
          generate: undefined
        }
      })
    } else {
      this.getDetailList()
    }
  },
  watch: {
    searchValue (newVal) {
      if (!newVal) {
        this.getDetailList()
      }
    }
  }
}
```
