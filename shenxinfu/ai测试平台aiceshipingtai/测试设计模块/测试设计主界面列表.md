```js
<template>
  <div class="test-design">
    <!-- Header布局 -->
    <a-layout-header class="ant-header-fixedHeader ant-header-side-closed" :style="{ padding: '0', zIndex: 10 }">
      <div class="design-header">
        <div class="header-title">
          <IxSpace size="md" align="center">
            <!-- 返回按钮 -->
            <IxButton class="return-icon-button" @click="returnFunc" mode="link">
              <IxIcon class="return-icon" name="arrow-left" />
            </IxButton>
            <!-- 设计名称容器 -->
            <div class="designName-container">
              <IxTooltip :title="`${designName}`">
                <div class="designName">{{ designName }}</div>
              </IxTooltip>
              <!-- 编辑名称按钮 -->
              <IxButton mode="link" class="edit-icon-button" @click="editName">
                <IxTooltip title="编辑名称">
                  <IxIcon class="edit-icon" name="edit" />
                </IxTooltip>
              </IxButton>
              <span class="separator">|</span>
            </div>
          </IxSpace>
        </div>
        <!-- 状态显示 -->
        <div v-if="testCaseState.task_step" class="status">
          <IxTooltip :title="`${testCaseState.task_step_display}阶段的AI生成状态：${testCaseState.task_step_state_display}`">
            <div class="status">
              <img :src="getImageSrc(testCaseState.task_step_state_display)" style="width: 16px" />
              <div style="width: 40px">
                {{ testCaseState.task_step_state_display }}
              </div>
            </div>
          </IxTooltip>
          <IxTooltip v-if="testCaseState.task_step_state_display === '失败'" :title="`${testCaseState.state_message}`">
            <img src="./imgs/question.svg" alt="question" style="width: 16px" />
          </IxTooltip>
        </div>
        <!-- 菜单栏 -->
        <a-menu mode="horizontal" :selectedKeys="[activeTab]">
          <a-menu-item key="1" @click="updateActiveTab('1')">
            <span class="menu-item">需求分析</span>
          </a-menu-item>
          <a-menu-item key="2" @click="updateActiveTab('2')">
            <span class="menu-item">测试分析</span>
          </a-menu-item>
          <a-menu-item key="3" @click="updateActiveTab('3')">
            <span class="menu-item">测试设计</span>
          </a-menu-item>
        </a-menu>
       <!-- 切换流程显示 -->
        <span class="toggle-process" @click="toggleProcessVisibility">
          {{ processVisible ? '隐藏测试设计使用流程' : '显示测试设计使用流程' }}
        </span>
      </div>
    </a-layout-header>
    <div class="design-content">
      <!-- 流程图片显示 -->
      <div v-if="processVisible" class="img-testprocess">
        <img class="process-img" src="./imgs/testprocess.svg" />
        <img class="close-icon" src="./imgs/icon.svg" @click="toggleProcessVisibility" />
      </div>
      <!-- 不同视图组件 -->
      <TestDemandView
        ref="TestDemandView"
        v-if="activeTab === '1'"
        :designId="designId"
        @update-demandtotal="handleUpdateDemandTotal"
        @update-active-tab="updateActiveTab"
      />
      <DesignAnalysisView
        ref="DesignAnalysisView"
        v-else-if="activeTab === '2'"
        :designId="designId"
        :demandtotal="demandtotal"
        @update-active-tab="updateActiveTab"
      />
      <TestPlanView
        ref="TestPlanView"
        v-else-if="activeTab === '3'"
        :designId="designId"
        :demandtotal="demandtotal"
        @update-active-tab="updateActiveTab"
      />
    </div>
    <!-- 编辑名称模态框 -->
    <IxModal :visible.sync="showEditModal" :width="600" header="编辑名称" :onOk="submit" :mask="false">
      <IxForm class="edit-form">
        <IxFormItem label="名称" message="请输入名称!" required :labelCol="4" :controlCol="20">
          <IxInput :value.sync="name"></IxInput>
        </IxFormItem>
      </IxForm>
    </IxModal>
    <!-- 确认离开模态框 -->
    <IxModal
      :visible.sync="confirmVisible"
      type="confirm"
      header="提示"
      okText="保存并离开"
      cancelText="离开"
      title="内容编辑未保存，是否离开？"
      :onOk="saveAndReturn"
      :onCancel="onlyReturn"
    >
      <p style="color: #6f7785" v-if="hasLastedData">
        在你编辑测试设计期间，有其他用户已对当前测试设计进行了变更，如继续保存将覆盖测试设计。
      </p>
    </IxModal>
  </div>
</template>
export default {
  name: 'AddDesign',
  components: {
    TestDemandView,
    DesignAnalysisView,
    TestPlanView
  },
  data () {
    return {
      confirmVisible: false,
      hasConfirm: false,
      showEditModal: false,
      name: '',
      hasLastedData: false,
      activeTab: '1',
      processVisible: true,
      changeTab: '',
      demandtotal: 0
    }
  },
  computed: {
    ...mapState({
      isReadonly: (state) => state.mindmap.isReadonly,
      testCaseState: (state) => state.mindmap.testCaseState
    }),
    designId () {
      return this.$route.params.design_id
    },
    productId () {
      return Number(this.$route.query.productId)
    },
    designName () {
      return this.$route.query.name
    },
    queryActiveTab () {
      return this.$route.query.activeTab
    },
    activeTabName () {
      const tabNames = {
        1: '需求分析',
        2: '测试分析',
        3: '测试设计'
      }
      return tabNames[this.activeTab] || '需求分析'
    }
  },
 async beforeRouteLeave (to, from, next) {
    let viewRefs = null
    if (this.activeTab === '1') {
      next()
      return
    } else if (this.activeTab === '2') {
      viewRefs = this.$refs.DesignAnalysisView
    } else {
      viewRefs = this.$refs.TestPlanView
    }
    if (this.hasConfirm) {
      next()
      return
    }
    this.confirmVisible = viewRefs.$refs.mindmap.checkChange()
    if (this.confirmVisible) {
      this.hasLastedData = await viewRefs.$refs.mindmap.checkLastedData()
    } else {
      next()
    }
  },
  beforeDestroy () {
    this.setTimer()
    this.setActiveSidebar('')
  },
  created () {
    const activeTab = this.queryActiveTab || localStorage.getItem('activeTab_testCase_' + this.designId)
    this.activeTab = activeTab || '1'
    this.setActiveTab(this.activeTab)
    localStorage.setItem(`activeTab_testCase_${this.designId}`, this.activeTab)
    paramsToUrl({ activeTab: this.activeTab })
  },
 mounted () {
    const processVisible = localStorage.getItem('processVisible')
    if (processVisible !== null) {
      this.processVisible = JSON.parse(processVisible)
    }
    this.setProcessVisible(this.processVisible)
  },
  methods: {
    ...mapMutations(['setActiveTab', 'setActiveSidebar', 'setProcessVisible', 'setTimer']),
    handleUpdateDemandTotal (demandtotal) {
      this.demandtotal = demandtotal
    },
    getImageSrc (state) {
      // 定义一个状态到图片路径的映射对象
      const stateToImageMap = {
        排队中: require('@/views/testdesign/imgs/PENDING.svg'), // 排队中状态对应的图片路径
        进行中: require('@/views/testdesign/imgs/STARTED.svg'), // 进行中状态对应的图片路径
        成功: require('@/views/testdesign/imgs/SUCCESS.svg'), // 成功状态对应的图片路径
        已中止: require('@/views/testdesign/imgs/REVOKED.svg'), // 已中止状态对应的图片路径
        失败: require('@/views/testdesign/imgs/FAILURE.svg') // 失败状态对应的图片路径
      }
      // 根据传入的状态返回对应的图片路径，如果没有匹配的状态则返回 '-'
      return stateToImageMap[state] || '-'
    },
   async saveAndReturn () {
      let viewRefs = null
      // 根据 activeTab 的值选择不同的视图引用
      if (this.activeTab === '2') {
        viewRefs = this.$refs.DesignAnalysisView
      } else {
        viewRefs = this.$refs.TestPlanView
      }
      // 如果 viewRefs 存在，则调用 mindmap 的 saveData 方法保存数据
      if (viewRefs) {
        await viewRefs.$refs.mindmap.saveData()
      }
      // 设置一个延时操作，延时 300 毫秒后执行
      setTimeout(() => {
        this.hasConfirm = true
        if (this.changeTab) {
          this.handleTab(this.changeTab)
        } else {
          // 跳转到 DesignHome 路由
          this.$router.push({
            path: '/DesignHome'
          })
        }
      }, 300)
    },
    onlyReturn () {
      this.hasConfirm = true
      if (this.changeTab) {
        this.handleTab(this.changeTab)
      } else {
        this.$router.push({
          path: '/DesignHome'
        })
      }
    },
    returnFunc () {
      this.$router.push({
        path: '/DesignHome'
      })
    },
 editName () {
      let viewRefs = ''
      if (this.activeTab === '2') {
        viewRefs = this.$refs.DesignAnalysisView
      } else {
        viewRefs = this.$refs.TestPlanView
      }
      if (viewRefs) {
        viewRefs.$refs.mindmap.removeNodeActive()
      }
      this.name = this.designName
      this.showEditModal = true
    },
    submit () {
      if (!this.name) {
        this.$message.error('请输入名称')
        return false
      }
      updateDesign(this.designId, {
        project_id: this.productId,
        name: this.name
      }).then((resp) => {
        if (resp.id) {
          this.$message.success('编辑成功')
          this.$router.replace({
            query: { name: resp.name, productId: this.productId }
          })
        }
      })
    },
   toggleProcessVisibility () {
      this.processVisible = !this.processVisible
      this.setProcessVisible(this.processVisible)
      localStorage.setItem('processVisible', this.processVisible)
    },
    async updateActiveTab (tab) {
      let viewRefs = null
      if (this.activeTab === '2') {
        viewRefs = this.$refs.DesignAnalysisView
      } else if (this.activeTab === '3') {
        viewRefs = this.$refs.TestPlanView
      }
      if (viewRefs) {
        this.confirmVisible = viewRefs.$refs.mindmap.checkChange()
        if (this.confirmVisible) {
          this.changeTab = tab
          this.hasLastedData = await viewRefs.$refs.mindmap.checkLastedData()
        } else {
          this.handleTab(tab)
        }
      } else {
        this.handleTab(tab)
      }
    },
    handleTab (tab) {
      this.activeTab = tab
      this.changeTab = ''
      this.setActiveTab(this.activeTab)
      this.setActiveSidebar('')
      paramsToUrl({ activeTab: this.activeTab })
      localStorage.setItem(`activeTab_testCase_${this.designId}`, tab)
    }
  }
}
```

3.详情-需求分析-测试需求_列表

 列表有“序号、需求项、需求描述、操作”4列；
 
 [序号]：
 
 [需求项]：显示需求项名称。内容全部展示，根据内容自适应行高。
 
 [需求描述]：显示需求描述。内容全部展示，根据内容自适应行高。
 
 [操作]：有“编辑、删除”2个操作按钮；
 
 点击编辑则出现编辑弹窗；
 
 点击删除则出现二次确认删除气泡框；
 
 点击确定按钮，则关闭气泡框，并删除该条需求项；
 
 点击取消按钮，则关闭气泡框，操作不生效。
 
 排序规则：按照创建时间倒序排列，最近创建的在最上方；
 
 列表支持分页；
